Implementation Details

Spring Security Configuration

사용자 인증 및 권한 관리를 위해 Spring Security를 적용했습니다. 폼 기반 로그인과 역할(ROLE) 기반 접근 제어를 구현했습니다.

SecurityConfig.java (My Implementation)
@Configuration
@EnableWebSecurity
public class SecurityConfig extends WebSecurityConfigurerAdapter {

    @Override
    protected void configure(HttpSecurity http) throws Exception {
        http
            .authorizeRequests()
                .antMatchers("/admin/**").hasRole("ADMIN")
                .antMatchers("/board/**", "/write/**").hasAnyRole("USER", "ADMIN")
                .antMatchers("/", "/login", "/register").permitAll()
            .and()
            .formLogin()
                .loginPage("/login")
                .defaultSuccessUrl("/")
                .failureUrl("/login?error=true")
            .and()
            .logout()
                .logoutSuccessUrl("/")
            .and()
            .rememberMe()
                .tokenValiditySeconds(86400);  // 로그인 상태 유지
    }
}

    $.ajax({
        url: '/api/check-id',
        type: 'POST',
        data: { userId: userId },
        success: function(result) {
            if (result.available) {
                $('#id-status').text('사용 가능한 아이디입니다.').css('color', 'green');
            } else {
                $('#id-status').text('이미 사용중인 아이디입니다.').css('color', 'red');
            }
        }
    });
}
